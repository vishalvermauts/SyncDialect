453: // Helper to get flag emoji from locale tag
454: fun getFlagEmoji(localeTag: String): String {
455:     val locale = Locale.forLanguageTag(localeTag)
456:     val countryCode = locale.country
457:     if (countryCode.isEmpty()) {
458:         // Fallbacks for languages without clear country codes in ML Kit
459:         return when (locale.language) {
460:             "en" -> "🇺🇸"
461:             "es" -> "🇪🇸"
462:             "fr" -> "🇫🇷"
463:             "de" -> "🇩🇪"
464:             "it" -> "🇮🇹"
465:             "ja" -> "🇯🇵"
466:             "ko" -> "🇰🇷"
467:             "zh" -> "🇨🇳"
468:             "hi" -> "🇮🇳"
469:             "ru" -> "🇷🇺"
470:             "ar" -> "🇸🇦"
471:             "pt" -> "🇧🇷"
472:             "tr" -> "🇹🇷"
473:             "ur" -> "🇵🇰"
474:             "nl" -> "🇳🇱"
475:             else -> "🌐"
476:         }
477:     }
478:     val firstLetter = Character.codePointAt(countryCode, 0) - 0x41 + 0x1F1E6
479:     val secondLetter = Character.codePointAt(countryCode, 1) - 0x41 + 0x1F1E6
480:     return String(Character.toChars(firstLetter)) + String(Character.toChars(secondLetter))
481: }

483: @OptIn(ExperimentalMaterial3Api::class)
484: @Composable
485: fun LanguagePacksCard(accentColor: Color, cardBackground: Color, cardBorder: Color) {
486:     var downloadedModels by remember { mutableStateOf<Set<String>>(emptySet()) }
487:     var downloadingModels by remember { mutableStateOf<Set<String>>(emptySet()) }
488:     var isChecking by remember { mutableStateOf(true) }
489:     var showAddDialog by remember { mutableStateOf(false) }
490:     var languageToDelete by remember { mutableStateOf<Triple<String, String, String>?>(null) }
491:     
492:     val allTags = TranslateLanguage.getAllLanguages()
493:     val modelManager = RemoteModelManager.getInstance()
494: 
495:     val checkModels = {
496:         modelManager.getDownloadedModels(TranslateRemoteModel::class.java)
497:             .addOnSuccessListener { models ->
498:                 downloadedModels = models.map { it.language }.toSet()
499:                 isChecking = false
500:             }
501:             .addOnFailureListener {
502:                 isChecking = false
503:             }
504:     }
505:     
506:     LaunchedEffect(Unit) {
507:         checkModels()
508:     }
509: 
510:     val allLanguages = allTags.map { tag ->
511:         val displayName = Locale.forLanguageTag(tag).displayLanguage.replaceFirstChar { it.uppercase() }
512:         val flag = getFlagEmoji(tag)
513:         Triple(displayName, tag, flag)
514:     }.sortedBy { it.first }
515: 
516:     val installed = allLanguages.filter { downloadedModels.contains(it.second) }
517:     val available = allLanguages.filter { !downloadedModels.contains(it.second) && !downloadingModels.contains(it.second) }.take(5) // Show top 5 for visual cleanliness, "Manage Packs" opens full list
518:
